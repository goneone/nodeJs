비동기함수, 콜백함수 관련~!

전역객체(window)에 속한 메서드에는 경고창을 띄워주는 alert 그리고 setTimeout이라는 메서드 등이 있음.

function run() {
    setTimeout(function() {
        var msg = "hello codesquad";
        console.log(msg);  //이 메시지는 즉시 실행되지 않습니다.
    }, 0);
    console.log("run function end");
}

console.log("start");
run();
console.log("end");
